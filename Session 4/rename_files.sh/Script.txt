#!/bin/bash
if [[ $# -ne 2 ]]
then 
echo "Usage:./rename_files.sh <extention> <new_name> "
exit 1
fi
extention=$1
new_name=$2
counter=1
for file in *.$extention
do
if [[ -f $file ]]
then 
new_filename="$new_name$counter.$extention"
mv "$file" "$new_filename"
counter=$(($counter+1))
fiÂ 
done
